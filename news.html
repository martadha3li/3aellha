// دالة الاشتراك الحي واللحظي (Subscription) المحدثة لمزامنة الأخبار الفورية function listenToNewsFeed() { onSnapshot(query(collection(db, "News"), orderBy("publishDate", "desc")), (snap) => { const feed = document.getElementById('instagramFeed'); const pinnedArea = document.getElementById('pinnedArea'); feed.innerHTML = ""; pinnedArea.innerHTML = ""; snap.forEach(docSnap => { const post = docSnap.data(); const postId = docSnap.id; // 1. التعامل الدقيق مع الأخبار المثبتة (pinned) لتثبيتها في أعلى الصفحة شريط ذهبي if (post.pinned === true || post.isPinned === true) { const pinCard = document.createElement('div'); pinCard.className = "pin-card"; pinCard.onclick = () => showPinnedFull(postId); pinCard.innerHTML = `📌 ${post.title} عرض ←`; pinnedArea.appendChild(pinCard); } // 2. صياغة كرت الخبر بهيئة منشور إنستغرام (صورة مربعة، أسفلها النص والتاريخ) const instaCard = document.createElement('div'); instaCard.className = "instagram-post"; // حساب وتحويل الطابع الزمني المحدث (Date.now) بدقة تامة لتنسيق سعودي عائلي مرن let formattedDate = "قبل قليل"; if (post.publishDate) { // فحص إذا كان الوقت مخزن كميلي ثانية مباشرة أو كنظام الفايربيس القديم const dateObj = typeof post.publishDate === 'number' ? new Date(post.publishDate) : (post.publishDate.seconds ? new Date(post.publishDate.seconds * 1000) : new Date(post.publishDate)); formattedDate = dateObj.toLocaleString('ar-SA', { hour: '2-digit', minute: '2-digit', day: 'numeric', month: 'short', year: 'numeric' }); } instaCard.innerHTML = `
ع
أخبار العائلة
${isAdmin ? `
⋮
` : ''}

أخبار العائلة ${post.content || post.body || post.title || ''}

${formattedDate}
`; feed.appendChild(instaCard); }); // إظهار أو إخفاء مساحة التثبيت تلقائياً بحسب المحتوى المخزن pinnedArea.style.display = pinnedArea.innerHTML === "" ? "none" : "block"; }); }
